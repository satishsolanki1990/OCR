w = np.zeros(d)
w_bar = np.zeros(d)
s = 1
iters = 15
it = 0
accuracy_train = []
accuracy_val = []
w_bars = []
while it < iters:
    for i in range(n):
        if y[i]*(np.transpose(w).dot(X[i])) <= 0:
            w += y[i]*X[i]
        w_bar = (s*w_bar+w)/(s+1)
        s += 1
    it += 1
    # Accuracies :
    accuracy_train.append((np.sign(X.dot(w_bar))*y[:,0]==1).sum()/n)
    accuracy_val.append((np.sign(X_val.dot(w_bar))*y_val[:,0]==1).sum()/n_val)
    w_bars.append(w_bar)
